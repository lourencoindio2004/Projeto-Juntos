cd /etc/apache2/sites-available
sudo touch about.conf

about.conf:

<VirtualHost *:80>
  ServerName www.aboutt.pt
  DocumentRoot /var/www/about
</VirtualHost>

<VirtualHost *:443>
  ServerName www.aboutt.pt
  DocumentRoot /var/www/about

  SSLEngine on
  SSLCertificateFile /etc/ssl/certs/ssl-cert-snakeoil.pem
  SSLCertificateKeyFile /etc/ssl/private/ssl-cert-snakeoil.key

</VirtualHost>

<Directory /var/www/about>
  AllowOverride All
  Require all granted
</Directory>


/----------------------------------------------------------/

sudo apachectl configtest
sudo a2ensite about
sudo systemctl reload apache2

sudo a2enmod ssl
sudo systemctl restart apache2

intranet.conf:

<VirtualHost *:80>
  ServerName intranet.aboutt.pt
  DocumentRoot /var/www2/intranet
</VirtualHost>

<VirtualHost *:443>
  ServerName intranet.aboutt.pt
  DocumentRoot /var/www2/intranet

  SSLEngine on
  SSLCertificateFile /etc/ssl/certs/ssl-cert-snakeoil.pem
  SSLCertificateKeyFile /etc/ssl/private/ssl-cert-snakeoil.key

</VirtualHost>

<Directory /var/www2/intranet>
  AllowOverride All
  AuthUserFile /etc/apache2/password_file
  AuthName "Zona Protegida"
  AuthType Basic

<RequireAll>
    Require ip 192.168.20.1
    Require user admin
</RequireAll>


/----------------------------------------------------------/

sudo apachectl configtest
sudo a2ensite intranet
sudo systemctl reload apache2
